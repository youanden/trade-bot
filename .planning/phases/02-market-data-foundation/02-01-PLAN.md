    2. Create `src/worker/core/simulation/prng.ts`:
    ```typescript
    export function createPrng(seed: number): () => number {
      let s = (seed || 1) >>> 0;
      return function () {
        s += 0x6d2b79f5;
        let t = s;
        t = Math.imul(t ^ (t >>> 15), t | 1);
        t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
        return ((t ^ (t >>> 14)) >>> 0) / 0x100000000;
      };
    }
    ```

    - Scenario parameter table (internal const):
      - bull: drift=+0.05, sigma=0.10
      - bear: drift=-0.05, sigma=0.10
      - flat: drift=0.00, sigma=0.04

      h. Returns `{ market, prices }`
