    ```typescript
    describe("BT-04: circuit breaker day reset across simulated days", () => {
      test("circuit breaker fires on day 1 after losing positions seeded, trading resumes on day 2", async () => {
        // 48 ticks at 30min each = 24 hours, starting noon day 1
        const TICK_INTERVAL = 30 * 60 * 1000; // 30 min
        const scenario = generateScenario({
          type: "flat",
          seed: 7,
          ticks: 48,
          tickIntervalMs: TICK_INTERVAL,
          startTime: "2024-06-01T12:00:00.000Z",
        });

        // Track circuit breaker state and trade activity per day
        let breakerFiredOnDay1 = false;
        let tradesBeforeBreaker = 0;
        let tradesOnDay2 = 0;
        let tickCount = 0;

        // Custom strategy: seeds losing positions on tick 3, then checks breaker on each tick
        const circuitBreakerStrategy = async (bot: any, env: any) => {
          tickCount++;
          const db = env.DB;  // The mock passes through the drizzle DB directly

          // Import schema and risk inline (already available via mock intercepts)
          const { positions } = await import("../../src/worker/core/db/schema");
          const { PortfolioRisk } = await import("../../src/worker/core/risk/portfolio");
          const { getLimitsForBot } = await import("../../src/worker/core/risk/limits");

          const risk = new PortfolioRisk(db, getLimitsForBot("market-maker"));
          const now = new Date().toISOString(); // Uses SimulatedDate → simulated time
          const today = now.split("T")[0];

          // On tick 3 (still day 1): seed closed losing positions that exceed maxDailyLoss (500)
          if (tickCount === 3) {
            // Insert a market row reference (marketId 1 already seeded by engine)
            db.insert(positions).values({
              botInstanceId: bot.config.dbBotId,
              marketId: 1,  // Seeded by engine
              platform: "polymarket",
              outcome: "yes",
              size: 100,
              avgEntry: 0.5,
              currentPrice: 0.0,
              unrealizedPnl: -600,  // Exceeds maxDailyLoss of 500
              status: "closed",
              closedAt: now,  // Closed "today" in simulated time (2024-06-01)
            }).run();
          }

          // Check circuit breaker
          const breached = await risk.isDailyLossBreached();

          if (breached && today === "2024-06-01") {
            breakerFiredOnDay1 = true;
            return;  // Strategy skips trading (mimics real strategy behavior)
          }

          // If not breached, record a trade
          await bot.recordTrade({
            marketId: 1,
            platform: "polymarket",
            side: "buy",
            outcome: "yes",
            price: 0.5,
            size: 1,
            reason: `tick-${tickCount}`,
          });

          if (today === "2024-06-01" && !breached) {
            tradesBeforeBreaker++;
          }
          if (today === "2024-06-02") {
            tradesOnDay2++;
          }
        };

        // Register custom strategy temporarily
        const { getStrategy } = await import("../../src/worker/bots/registry");

        // We can't easily register a custom strategy in the registry,
        // so instead we use mock.module to intercept getStrategy for this test.
        // BUT mock.module is already set up at module level for db/factory.
        // Alternative: use the botType "market-maker" but override what runBacktest calls.
        //
        // Actually, the cleanest approach is to NOT use runBacktest for this test.
        // Instead, replicate the essential engine flow manually so we have full control.
        // This is an integration test of the circuit-breaker behavior, not the engine API.

        const { BacktestClock } = await import("../../src/worker/core/simulation/engine");
        const { SimulatedBot } = await import("../../src/worker/core/simulation/sim-bot");
        const { PriceFeed } = await import("../../src/worker/core/simulation/feed");
        const { SimExchangeClient } = await import("../../src/worker/core/simulation/sim-client");
        const { markets: marketsTable, prices: pricesTable, botInstances: botInstancesTable, positions: positionsTable } = await import("../../src/worker/core/db/schema");

        // Create isolated DB (same as engine does)
        const db = createTestDb();

        // Seed scenario data
        const [insertedMarket] = db.insert(marketsTable).values({
          platform: scenario.market.platform,
          platformId: scenario.market.platformId,
          title: scenario.market.title,
          status: scenario.market.status ?? "active",
        }).returning().all();

        for (const priceRow of scenario.prices) {
          db.insert(pricesTable).values({
            marketId: insertedMarket.id,
            yesPrice: priceRow.yesPrice,
            noPrice: priceRow.noPrice,
            timestamp: priceRow.timestamp,
          }).run();
        }

        const [insertedBot] = db.insert(botInstancesTable).values({
          botType: "market-maker",
          name: "test-breaker",
          status: "running",
          config: makeMarketMakerConfig(),
        }).returning().all();

        const botConfig = { ...makeMarketMakerConfig(), dbBotId: insertedBot.id };
        const clock = new BacktestClock("2024-06-01T12:00:00.000Z", TICK_INTERVAL);
        const feed = new PriceFeed(scenario);
        const bot = new SimulatedBot(botConfig as any, db);
        const simClient = new SimExchangeClient({
          platform: "polymarket",
          feed,
          simulatedNow: () => clock.now(),
          virtualBalance: 1000,
        });

        const env: any = {
          DB: db,
          _simClient: simClient,
          ENVIRONMENT: "backtest",
        };

        // Override Date for simulated time
        const OriginalDate = globalThis.Date;
        const SimulatedDate: any = function SimulatedDate(this: any, ...args: any[]) {
          if (args.length === 0) return new OriginalDate(clock.now());
          return new OriginalDate(...(args as [any]));
        };
        SimulatedDate.now = () => new OriginalDate(clock.now()).getTime();
        SimulatedDate.parse = OriginalDate.parse.bind(OriginalDate);
        SimulatedDate.UTC = OriginalDate.UTC.bind(OriginalDate);
        SimulatedDate.prototype = OriginalDate.prototype;

        globalThis.Date = SimulatedDate;
        try {
          for (let i = 0; i < scenario.prices.length; i++) {
            await circuitBreakerStrategy(bot, env);
            clock.advance();
          }
        } finally {
          globalThis.Date = OriginalDate;
        }

        // ASSERTIONS:
        // 1. Circuit breaker fired on day 1
        expect(breakerFiredOnDay1).toBe(true);

        // 2. Some trades happened before the breaker (ticks 1-2 on day 1)
        expect(tradesBeforeBreaker).toBeGreaterThan(0);

        // 3. Trades resumed on day 2 (circuit breaker reset after midnight)
        expect(tradesOnDay2).toBeGreaterThan(0);

        // 4. Total trade count = tradesBeforeBreaker + tradesOnDay2
        //    (no trades during breaker period on day 1)
        expect(bot._tradeCount).toBe(tradesBeforeBreaker + tradesOnDay2);
      });
    });
    ```

    **Important implementation notes:**

    1. The test manually replicates the engine's tick loop (DB creation, scenario seeding, Date override, tick loop, cleanup) rather than calling `runBacktest()`. This is necessary because the test needs a custom strategy function that is not registered in the strategy registry. This is acceptable for an integration test of the circuit-breaker behavior.

    2. The custom strategy function `circuitBreakerStrategy`:
       - On tick 3: inserts a closed position with `unrealizedPnl: -600` and `closedAt` set to simulated "now" (which is 2024-06-01 due to Date override). The -600 exceeds market-maker's `maxDailyLoss` of 500.
       - On every tick: instantiates `PortfolioRisk` (with default clockFn, which uses `new Date()` which is SimulatedDate) and calls `isDailyLossBreached()`.
       - If breached and still day 1: sets `breakerFiredOnDay1 = true` and returns without trading.
       - If not breached: calls `bot.recordTrade(...)` and increments the appropriate counter.

    3. Day boundary: ticks 0-23 are 2024-06-01 (12:00 to 23:30), ticks 24-47 are 2024-06-02 (00:00 to 11:30). The breaker fires after tick 3 and blocks remaining day-1 ticks. On tick 24 (first day-2 tick), `isDailyLossBreached()` checks for positions closed on "2024-06-02" and finds none, so it returns false and trading resumes.

    4. The `PortfolioRisk` import uses dynamic import since it is a real module (not mocked). The `positions` schema import also uses dynamic import for consistency.

    5. Make sure `positions` import is available for the `.insert()` call. The `db` variable is the bun-sqlite drizzle instance from `createTestDb()` — use `.run()` for inserts (synchronous).

    6. Keep the existing test file structure intact. Only replace the `describe("BT-04: ...")` block (lines 222-271). Do not modify any other test blocks.
